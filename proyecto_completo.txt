class PlaceholderEntry(ttk.Entry):
    def __init__(self, container, placeholder, color='grey', *args, **kwargs):
        super().__init__(container, *args, **kwargs)
        self.placeholder = placeholder
        self.placeholder_color = color
        self.default_fg_color = 'black'
        self.is_placeholder = False
        self.bind("<FocusIn>", self._clear_placeholder)
        self.bind("<FocusOut>", self._add_placeholder)
        self._add_placeholder()

    def _clear_placeholder(self, event=None):
        if self.is_placeholder:
            self.delete(0, 'end')
            self.config(foreground=self.default_fg_color)
            self.is_placeholder = False

    def _add_placeholder(self, event=None):
        if not super().get():
            self.insert(0, self.placeholder)
            self.config(foreground=self.placeholder_color)
            self.is_placeholder = True

    def get(self):
        value = super().get()
        return "" if self.is_placeholder else value

import requests
import xml.etree.ElementTree as ET
from Utils.logger import get_logger
from config import ALLOWED_UPDATE_TYPES

# Deshabilitar warnings SSL para entornos de laboratorio/pruebas.
requests.packages.urllib3.disable_warnings()

# Instanciamos el logger para este mÃ³dulo con el nombre del mÃ³dulo actual.
logger = get_logger(__name__)


def _send_request(url, params, timeout=30):
    """
    Envia peticiÃ³n HTTP GET a la API de Palo Alto con los parÃ¡metros dados,
    devuelve la raÃ­z XML parseada o None en caso de error.
    Controla excepciones y errores HTTP.
    """
    try:
        response = requests.get(url, params=params, verify=False, timeout=timeout)
        if response.status_code != 200:
            logger.error(f"HTTP error {response.status_code} en request: {response.text[:300]}")
            return None
        try:
            return ET.fromstring(response.text)
        except ET.ParseError:
            logger.error(f"Error parsing XML: {response.text[:300]}")
            return None
    except Exception as e:
        logger.error(f"ExcepciÃ³n en request API: {e}")
        return None


def download_dynamic_update(firewall_ip: str, api_key: str, update_type: str) -> bool:
    """
    Consulta y descarga la Ãºltima actualizaciÃ³n dinÃ¡mica disponible para un tipo especÃ­fico.

    1) Valida que el tipo de actualizaciÃ³n estÃ© permitido.
    2) Consulta la Ãºltima versiÃ³n disponible usando la API XML.
    3) Si hay versiÃ³n disponible, inicia la descarga.
    4) Registra cada paso en el logger para trazabilidad.
    5) Maneja errores y excepciones de conexiÃ³n y formato XML.

    Nota: Si el firewall responde que el update no estÃ¡ disponible (status="error"),
    la funciÃ³n registrarÃ¡ el error y devolverÃ¡ False â€” **sin** interrumpir el flujo.
    """

    # Validar primero que el tipo de actualizaciÃ³n es soportado (lista central en config).
    if update_type not in ALLOWED_UPDATE_TYPES:
        logger.error(f"Tipo de actualizaciÃ³n no permitido (no estÃ¡ en ALLOWED_UPDATE_TYPES): {update_type}")
        return False

    # Construir la URL base para la API
    url = f"https://{firewall_ip}/api/"
    logger.info(f"Verificando actualizaciÃ³n para: {update_type}")

    # ParÃ¡metros para consultar la Ãºltima versiÃ³n disponible de la actualizaciÃ³n
    check_params = {
        'type': 'op',
        'cmd': f'<request><{update_type}><upgrade><check/></upgrade></{update_type}></request>',
        'key': api_key,
    }

    # Enviar peticiÃ³n de consulta y parsear XML
    root_check = _send_request(url, check_params)
    if root_check is None:
        logger.error(f"Fallo al consultar versiones disponibles para {update_type} (respuesta nula o invÃ¡lida).")
        return False

    # Si la API responde con status != success, es probable que el update no exista/haya sido deshabilitado.
    if root_check.attrib.get('status') != 'success':
        # Extraer mensaje legible (si existe) y registrar como "no disponible"
        msg = root_check.findtext('./msg/line') or root_check.findtext('.//line') or root_check.findtext('./msg') or 'No disponible'
        logger.error(f"Dynamic Update '{update_type}' NO disponible en el firewall: {msg}")
        return False

    # Buscar las entradas de versiones disponibles
    versions = root_check.findall('.//entry')
    if not versions:
        logger.warning(f"No hay actualizaciones disponibles para {update_type}")
        return False

    # Extraer la primera/Ãºltima versiÃ³n (segÃºn estructura)
    latest_version_node = versions[0].find('version')
    if latest_version_node is None or not latest_version_node.text:
        logger.warning(f"No se pudo obtener versiÃ³n vÃ¡lida para {update_type}")
        return False

    latest_version = latest_version_node.text.strip()
    logger.info(f"Ãšltima versiÃ³n para {update_type}: {latest_version}")

    # ParÃ¡metros para iniciar la descarga de la Ãºltima versiÃ³n
    download_params = {
        'type': 'op',
        'cmd': f'<request><{update_type}><upgrade><download><latest/></download></upgrade></{update_type}></request>',
        'key': api_key,
    }

    logger.info(f"Descargando actualizaciÃ³n {update_type} versiÃ³n {latest_version}")

    # Enviar peticiÃ³n para descarga, timeout extendido por ser operaciÃ³n pesada
    root_download = _send_request(url, download_params, timeout=300)
    if root_download is None:
        logger.error(f"Fallo la descarga para {update_type} (respuesta nula o invÃ¡lida).")
        return False

    # Si la respuesta indica error, registrar y devolver False (sin romper el proceso).
    if root_download.attrib.get('status') != 'success':
        msg = root_download.findtext('./msg/line') or root_download.findtext('.//line') or 'Error en descarga'
        logger.error(f"Error al descargar {update_type}: {msg}")
        return False

    logger.info(f"ActualizaciÃ³n {update_type} versiÃ³n {latest_version} descargada correctamente")
    return True

def download_panos_version(firewall_ip: str, api_key: str, version_to_download: str) -> bool:
    """
    Descarga la versiÃ³n especÃ­fica de PAN-OS al firewall (sin instalarla).

    Proceso:
    1. Consulta versiones disponibles para refrescar el catÃ¡logo.
    2. Inicia la descarga de la versiÃ³n solicitada usando la API.
    3. Maneja errores y registra cada paso.

    Args:
        firewall_ip (str): IP del firewall Palo Alto.
        api_key (str): Clave API para autenticaciÃ³n.
        version_to_download (str): VersiÃ³n PAN-OS a descargar (ej. "10.2.8").

    Returns:
        bool: True si la descarga inicia con Ã©xito, False si falla.
    """

    url = f"https://{firewall_ip}/api/"
    logger.info("Iniciando consulta de versiones disponibles (puede tardar)...")

    check_params = {
        'type': 'op',
        'cmd': '<request><system><software><check></check></software></system></request>',
        'key': api_key
    }

    try:
        # Paso 1: consulta versiones disponibles
        check_response = requests.get(url, params=check_params, verify=False, timeout=60)
        if check_response.status_code != 200:
            logger.error(f"Error HTTP al consultar versiones disponibles: cÃ³digo {check_response.status_code}")
            return False

        check_root = ET.fromstring(check_response.text)
        if check_root.attrib.get('status') != 'success':
            logger.warning("Advertencia: la consulta de versiones no fue exitosa, seguirÃ¡ con la descarga...")
            # logger.debug(check_response.text)  # Para depuraciÃ³n

        # Paso 2: iniciar descarga
        logger.info(f"Iniciando descarga de PAN-OS versiÃ³n {version_to_download} (sin instalar)...")
        download_params = {
            'type': 'op',
            'cmd': f'<request><system><software><download><version>{version_to_download}</version></download></software></system></request>',
            'key': api_key
        }
        download_response = requests.get(url, params=download_params, verify=False, timeout=300)

        root_download = ET.fromstring(download_response.text)
        status = root_download.attrib.get('status')

        if status == 'success':
            logger.info(f"VersiÃ³n {version_to_download} puesta en descarga correctamente.")
            return True
        else:
            logger.error(f"Error intentando descargar versiÃ³n {version_to_download}: {download_response.text}")
            return False

    except requests.exceptions.RequestException as e:
        logger.error(f"Error de conexiÃ³n al firewall durante operaciÃ³n de software: {e}")
        return False
    except ET.ParseError as e:
        logger.error(f"Error al analizar XML de operaciÃ³n de software: {e}")
        return False
    except Exception as e:
        logger.error(f"Error inesperado durante operaciÃ³n de software: {e}")
        return False

def get_firewall_dashboard_data(firewall_ip: str, api_key: str) -> dict:
    """
    Recopila informaciÃ³n del dashboard (show system info) de un firewall Palo Alto Networks.

    Realiza una llamada a la API operativa para obtener datos generales del sistema,
    como hostname, modelo, serial, versiÃ³n, uptime e IP de gestiÃ³n.

    Args:
        firewall_ip (str): IP del firewall Palo Alto.
        api_key (str): Clave API para autenticaciÃ³n.

    Returns:
        dict: Diccionario con la informaciÃ³n del dashboard:
              - 'hostname': nombre del dispositivo
              - 'model': modelo del firewall
              - 'serial': nÃºmero de serie
              - 'version': versiÃ³n del software
              - 'uptime': tiempo de actividad del dispositivo
              - 'mgmt_ip': IP de gestiÃ³n
              Retorna diccionario vacÃ­o si ocurre algÃºn error.
    """
    url = f"https://{firewall_ip}/api/"
    params = {
        'type': 'op',
        'cmd': '<show><system><info/></system></show>',
        'key': api_key,
    }

    try:

import requests

def safe_findtext(element, tag, default=''):
    """
    Retorna el texto limpio del hijo 'tag' si estÃ¡ presente, o el valor 'default' si no.
    Evita errores al extraer datos XML con nodos faltantes o vacÃ­os.
    """
    if element is None:
        return default
    text = element.findtext(tag)
    if text is None:
        return default
    return text.strip()

def update_interface_data(data_dict, name, updates):
    """
    Actualiza el diccionario 'data_dict' para la interfaz 'name' con los valores de 'updates'.
    No actualiza si 'name' estÃ¡ vacÃ­o ni si los valores son None o vacÃ­os.
    """
    if not name:
        return
    for key, value in updates.items():
        if value:
            data_dict[name][key] = value

def get_firewall_interfaces_data(firewall_ip: str, api_key: str) -> dict:
    """
    Recopila informaciÃ³n detallada de interfaces de un firewall Palo Alto Networks.

    Realiza dos llamadas a la API:
    1. Estado operativo de interfaces: nombre, estado, MAC, zona, VR y tipo.
    2. ConfiguraciÃ³n para obtener perfil de gestiÃ³n (MGMT Profile).

    Args:
        firewall_ip (str): IP del firewall.
        api_key (str): Clave API para autenticaciÃ³n.

    Returns:
        dict: Diccionario donde la clave es el nombre de la interfaz y el valor
              es otro diccionario con detalles de la interfaz. Retorna {} si falla.
    """
    all_interfaces_data = defaultdict(lambda: {
        'name': '',
        'state': '',
        'ip': '',
        'mac': '',
        'zone': '',
        'mgmt_profile': 'N/A',
        'virtual_router': '',
        'interface_type': ''
    })

    try:
        # --- Primera llamada: estado operativo de interfaces ---
        url_op = f"https://{firewall_ip}/api/"
        params_op = {
            'type': 'op',
            'cmd': '<show><interface>all</interface></show>',
            'key': api_key
        }

        logger.info("Inicio llamada API: estado operativos de interfaces")
        response_op = requests.get(url_op, params=params_op, verify=False, timeout=10)

        if response_op.status_code != 200:
            logger.error(f"Error HTTP en estado interfaces: {response_op.status_code}")
            return {}

        root_op = ET.fromstring(response_op.text)
        if root_op.attrib.get('status') != 'success':
            logger.error("La respuesta API para estado interfaces no fue exitosa")
            logger.debug(response_op.text)
            return {}

        # --- Procesar secciÃ³n <hw>: nombre, estado, MAC ---
        for entry in root_op.findall('.//hw/entry'):
            name = safe_findtext(entry, 'name')
            update_interface_data(all_interfaces_data, name, {
                'name': name,
                'state': safe_findtext(entry, 'state'),
                'mac': safe_findtext(entry, 'mac')
            })

        # --- Procesar secciÃ³n <ifnet>: zona, IP, VR y tipo de interfaz ---
        for entry in root_op.findall('.//ifnet/entry'):
            name = safe_findtext(entry, 'name')
            zone = safe_findtext(entry, 'zone')
            ip = safe_findtext(entry, 'ip')
            fwd_text = safe_findtext(entry, 'fwd')

            vr = ''
            if fwd_text.startswith('vr:'):
                vr = fwd_text.replace('vr:', '')
            elif fwd_text == 'N/A':
                vr = 'N/A'

            # Determinar tipo de interfaz segÃºn nombre e IP
            interface_type = ''
            if 'ethernet' in name.lower() or name.startswith('ae'):
                interface_type = 'Layer3' if ip and ip != 'N/A' else 'Layer2/Physical'
            elif name.startswith('vlan'):
                interface_type = 'VLAN'
            elif name.startswith('loopback'):
                interface_type = 'Loopback'
            elif name.startswith('tunnel'):
                interface_type = 'Tunnel'
            elif name.startswith('sdwan'):
                interface_type = 'SD-WAN/AE'
            else:
                interface_type = 'Otro'

            update_interface_data(all_interfaces_data, name, {
                'name': name,
                'zone': zone,
                'ip': ip if ip and ip != 'N/A' else '',
                'virtual_router': vr,
                'interface_type': interface_type
            })

        # --- Segunda llamada: configuraciÃ³n para perfiles de gestiÃ³n (MGMT Profile) ---
        url_config = f"https://{firewall_ip}/api/"
        params_config = {
            'type': 'config',
            'action': 'get',
            'xpath': "/config/devices/entry[@name='localhost.localdomain']/network/interface",
            'key': api_key
        }

        logger.info("Inicio llamada API: configuraciÃ³n interfaces para MGMT Profiles")
        response_config = requests.get(url_config, params=params_config, verify=False, timeout=10)

        if response_config.status_code != 200:
            logger.error(f"Error HTTP en configuraciÃ³n interfaces: {response_config.status_code}")
            return {}

        root_config = ET.fromstring(response_config.text)
        if root_config.attrib.get('status') != 'success':
            logger.error("Respuesta API para configuraciÃ³n interfaces no fue exitosa")
            logger.debug(response_config.text)
        else:
            # Actualizar MGMT Profile en el diccionario de interfaces
            for entry in root_config.findall('.//interface//entry'):
                name = entry.attrib.get('name', '').strip()
                if not name:
                    continue
                mgmt_profile_node = entry.find('.//interface-management-profile')
                if mgmt_profile_node is not None and mgmt_profile_node.text:
                    if name in all_interfaces_data:
                        all_interfaces_data[name]['mgmt_profile'] = mgmt_profile_node.text.strip()

        logger.info(f"Recopiladas {len(all_interfaces_data)} interfaces")

        return all_interfaces_data

    except requests.exceptions.RequestException as e:
        logger.error(f"Error de conexiÃ³n con el firewall: {e}")
        return {}

    except ET.ParseError as e:
        logger.error(f"Error al analizar XML de la API: {e}")
        return {}

    except Exception as e:
        logger.error(f"Error inesperado en la recopilaciÃ³n de datos: {e}")
        return {}

def get_vpn_config(firewall_ip: str, api_key: str) -> list:
    """Obtiene la configuraciÃ³n de tÃºneles IPsec."""
    params = {
        'type': 'config',
        'action': 'get',
        'xpath': "/config/devices/entry[@name='localhost.localdomain']/network/tunnel",
        'key': api_key,
    }
    root = _api_get_request(firewall_ip, params)
    tunnels = []

    if root is None or root.attrib.get('status') != 'success':
        logger.error("Error al obtener configuraciÃ³n de tÃºneles VPN")
        return tunnels

    # Cache de IKE Gateways ya procesados
    ike_gateway_cache = {}

    for tun_entry in root.findall("./result/tunnel/ipsec/entry"):
        tunnel_name = tun_entry.get('name', NA)
        if tunnel_name == NA:
            continue

        # IKE Gateway
        ike_entry = tun_entry.find('auto-key/ike-gateway/entry')
        ike_name = ike_entry.get('name') if ike_entry is not None else NA

        # IPsec Profile
        ipsec_profile = NA
        ipsec_node = tun_entry.find('auto-key/ipsec-vpn')
        if ipsec_node is not None:
            ipsec_profile = ipsec_node.findtext('ipsec-crypto-profile', NA)

        tunnel_interface = tun_entry.findtext('tunnel-interface', NA)

        tunnel_info = {
            'name': tunnel_name,
            'ike_gateway_name': ike_name,
            'ipsec_profile': ipsec_profile,
            'tunnel_interface': tunnel_interface,
            'ike_local_ip': NA,
            'ike_local_interface': NA,
            'ike_peer_address_config': NA,
        }

        # Detalles IKE Gateway (cache)
        if ike_name != NA:
            if ike_name not in ike_gateway_cache:
                ike_gateway_cache[ike_name] = _get_ike_gateway_details(firewall_ip, api_key, ike_name)
            tunnel_info.update(ike_gateway_cache[ike_name])

        tunnels.append(tunnel_info)

    logger.info(f"TÃºneles VPN configurados: {[t['name'] for t in tunnels]}")
    return tunnels

def _parse_sa(root, sa_type='ike') -> dict:
    """Parsea XML de SA IKE o IPsec."""
    sas = {}
    if root is None or root.attrib.get('status') != 'success':
        return sas

    for entry in root.findall(".//entry"):
        if sa_type == 'ike':
            gw_name = entry.findtext(TAG_NAME, NA)
            state = entry.findtext(TAG_STATE, STATE_ACTIVE)
            peer = entry.findtext(TAG_PEER_ADDRESS, NA)
            if gw_name != NA:
                sas[gw_name] = {'state': state, 'peer_address': peer}
        else:
            tunnel_name = entry.findtext('tunnel-name') or entry.findtext(TAG_NAME, NA)
            state = entry.findtext(TAG_STATE, STATE_ACTIVE)
            life_time = entry.findtext(TAG_LIFE_TIME, NA)
            life_bytes = entry.findtext(TAG_LIFE_BYTES, NA)
            if tunnel_name != NA:
                sas[tunnel_name] = {'state': state, 'life_time': life_time, 'life_bytes': life_bytes}
    return sas

def get_vpn_status(firewall_ip: str, api_key: str) -> tuple:
    """Obtiene estado operacional IKE y IPsec SA."""
    params_ike = {'type': 'op', 'cmd': '<show><vpn><ike-sa/></vpn></show>', 'key': api_key}
    params_ipsec = {'type': 'op', 'cmd': '<show><vpn><ipsec-sa/></vpn></show>', 'key': api_key}

    ike_root = _api_get_request(firewall_ip, params_ike, timeout=20)
    ipsec_root = _api_get_request(firewall_ip, params_ipsec, timeout=20)

    ike_data = _parse_sa(ike_root, 'ike')
    ipsec_data = _parse_sa(ipsec_root, 'ipsec')
    return ike_data, ipsec_data

def correlate_vpn_status(configured_tunnels: list, ike_data: dict, ipsec_data: dict) -> list:
    """Combina configuraciÃ³n y estado operativo para cada tÃºnel."""
    full_report = []

    for tunnel in configured_tunnels:
        name = tunnel['name']
        ike_name = tunnel.get('ike_gateway_name')

        ike_status = STATUS_DOWN
        ipsec_status = STATUS_DOWN

        if ike_name and ike_name in ike_data:
            ike_status = STATUS_UP if ike_data[ike_name]['state'] == STATE_ACTIVE else STATUS_DOWN

        # Buscar matching IPsec SA
        ipsec_key = name if name in ipsec_data else next((k for k in ipsec_data if k.startswith(f"{name}:")), None)
        if ipsec_key:
            ipsec_status = STATUS_UP if ipsec_data[ipsec_key]['state'] == STATE_ACTIVE else STATUS_DOWN
            tunnel['ipsec_life_time'] = ipsec_data[ipsec_key].get('life_time', NA)
            tunnel['ipsec_life_bytes'] = ipsec_data[ipsec_key].get('life_bytes', NA)
        else:
            tunnel['ipsec_life_time'] = NA
            tunnel['ipsec_life_bytes'] = NA

        tunnel['ike_status'] = ike_status
        tunnel['ipsec_tunnel_state'] = ipsec_status
        tunnel['overall_status'] = STATUS_UP if ike_status == STATUS_UP and ipsec_status == STATUS_UP else STATUS_DOWN
        tunnel['ike_peer_ip'] = tunnel.get('ike_peer_address_config', NA)

        full_report.append(tunnel)
        logger.info(f"TÃºnel '{name}': Overall={tunnel['overall_status']}, IKE={ike_status}, IPsec={ipsec_status}")

    return full_report

    """

        'type': 'op',
        'cmd': '<request><tech-support><dump></dump></tech-support></request>',
        'key': api_key
    }

        response = requests.get(url, params=params, verify=False, timeout=600)  # Timeout extendido por operaciÃ³n pesada

            root = ET.fromstring(response.text)
            status = root.get('status')

                logger.info("Archivo TFS en proceso de generaciÃ³n en el firewall.")
                logger.info("Puedes descargarlo luego desde la GUI o mediante otro script si lo deseas.")
                return True
            else:
                logger.error("La solicitud de generaciÃ³n de TFS no fue exitosa. Revisar salida XML.")
                logger.debug(response.text)
                return False
        else:
            logger.error(f"Error HTTP al solicitar generaciÃ³n de TFS: CÃ³digo {response.status_code}")
            logger.debug(f"Respuesta parcial del firewall: {response.text[:500]}")
            return False

    except requests.exceptions.RequestException as e:
        logger.error(f"Error de conexiÃ³n al solicitar TFS: {e}")
        return False
    except ET.ParseError as e:
        logger.error(f"Error al analizar XML de la respuesta TFS: {e}")
        return False
    except Exception as e:
        logger.error(f"Error inesperado al solicitar generaciÃ³n de TFS: {e}")
        return False

def _get_value_or_na(value):

def apply_zebra_style(table_style, num_rows):
    for row in range(1, num_rows):

import os
import logging
from logging.handlers import RotatingFileHandler

LOG_DIR = os.path.join(os.path.dirname(os.path.abspath(__file__)), '..', 'Logs')
if not os.path.exists(LOG_DIR):
    os.makedirs(LOG_DIR)

LOG_FILE = os.path.join(LOG_DIR, 'paloalto.log')

def get_logger(name="paloalto"):
    logger = logging.getLogger(name)
    logger.setLevel(logging.INFO)
    if not logger.hasHandlers():
        # Console handler
        ch = logging.StreamHandler()
        ch.setLevel(logging.INFO)
        formatter = logging.Formatter('%(asctime)s [%(levelname)s] %(message)s')
        ch.setFormatter(formatter)
        logger.addHandler(ch)

        # File handler; by now the folder definitely exists
        fh = RotatingFileHandler(LOG_FILE, maxBytes=10**6, backupCount=5, encoding='utf-8')
        fh.setLevel(logging.INFO)
        fh.setFormatter(formatter)
        logger.addHandler(fh)
    return logger
